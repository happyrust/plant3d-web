# 校审系统数据库设计

## SurrealDB 表结构

### 1. users 表 - 用户

```surql
-- 用户表
DEFINE TABLE users SCHEMAFULL;

DEFINE FIELD username ON users TYPE string ASSERT $value != NONE;
DEFINE FIELD email ON users TYPE string ASSERT $value != NONE;
DEFINE FIELD name ON users TYPE string ASSERT $value != NONE;
DEFINE FIELD avatar ON users TYPE option<string>;
DEFINE FIELD role ON users TYPE string ASSERT $value IN ['admin', 'manager', 'reviewer', 'proofreader', 'designer', 'viewer'];
DEFINE FIELD department ON users TYPE option<string>;
DEFINE FIELD phone ON users TYPE option<string>;
DEFINE FIELD status ON users TYPE string DEFAULT 'active' ASSERT $value IN ['active', 'inactive', 'suspended'];
DEFINE FIELD password_hash ON users TYPE string;
DEFINE FIELD created_at ON users TYPE datetime DEFAULT time::now();
DEFINE FIELD updated_at ON users TYPE datetime DEFAULT time::now();
DEFINE FIELD last_login_at ON users TYPE option<datetime>;

-- 索引
DEFINE INDEX users_username_idx ON users FIELDS username UNIQUE;
DEFINE INDEX users_email_idx ON users FIELDS email UNIQUE;
DEFINE INDEX users_role_idx ON users FIELDS role;
DEFINE INDEX users_status_idx ON users FIELDS status;
```

### 2. review_tasks 表 - 编校审单/审核任务

```surql
-- 编校审单/审核任务表
DEFINE TABLE review_tasks SCHEMAFULL;

DEFINE FIELD title ON review_tasks TYPE string ASSERT $value != NONE;
DEFINE FIELD description ON review_tasks TYPE string DEFAULT '';
DEFINE FIELD model_name ON review_tasks TYPE string ASSERT $value != NONE;
DEFINE FIELD status ON review_tasks TYPE string DEFAULT 'draft'
    ASSERT $value IN ['draft', 'submitted', 'in_review', 'approved', 'rejected', 'cancelled'];
DEFINE FIELD priority ON review_tasks TYPE string DEFAULT 'medium'
    ASSERT $value IN ['low', 'medium', 'high', 'urgent'];
DEFINE FIELD requester_id ON review_tasks TYPE record<users>;
DEFINE FIELD requester_name ON review_tasks TYPE string;
DEFINE FIELD reviewer_id ON review_tasks TYPE record<users>;
DEFINE FIELD reviewer_name ON review_tasks TYPE string;
DEFINE FIELD components ON review_tasks TYPE array DEFAULT [];
DEFINE FIELD attachments ON review_tasks TYPE option<array>;
DEFINE FIELD review_comment ON review_tasks TYPE option<string>;
DEFINE FIELD due_date ON review_tasks TYPE option<datetime>;
DEFINE FIELD created_at ON review_tasks TYPE datetime DEFAULT time::now();
DEFINE FIELD updated_at ON review_tasks TYPE datetime DEFAULT time::now();

-- 索引
DEFINE INDEX review_tasks_status_idx ON review_tasks FIELDS status;
DEFINE INDEX review_tasks_priority_idx ON review_tasks FIELDS priority;
DEFINE INDEX review_tasks_requester_idx ON review_tasks FIELDS requester_id;
DEFINE INDEX review_tasks_reviewer_idx ON review_tasks FIELDS reviewer_id;
DEFINE INDEX review_tasks_created_idx ON review_tasks FIELDS created_at;
```

### 3. review_records 表 - 审核确认记录

```surql
-- 审核确认记录表（批注和测量数据快照）
DEFINE TABLE review_records SCHEMAFULL;

DEFINE FIELD task_id ON review_records TYPE record<review_tasks>;
DEFINE FIELD type ON review_records TYPE string DEFAULT 'batch';
DEFINE FIELD annotations ON review_records TYPE array DEFAULT [];
DEFINE FIELD cloud_annotations ON review_records TYPE array DEFAULT [];
DEFINE FIELD rect_annotations ON review_records TYPE array DEFAULT [];
DEFINE FIELD obb_annotations ON review_records TYPE array DEFAULT [];
DEFINE FIELD measurements ON review_records TYPE array DEFAULT [];
DEFINE FIELD note ON review_records TYPE string DEFAULT '';
DEFINE FIELD confirmed_by ON review_records TYPE record<users>;
DEFINE FIELD confirmed_at ON review_records TYPE datetime DEFAULT time::now();

-- 索引
DEFINE INDEX review_records_task_idx ON review_records FIELDS task_id;
DEFINE INDEX review_records_confirmed_at_idx ON review_records FIELDS confirmed_at;
```

### 4. review_history 表 - 审核历史日志

```surql
-- 审核历史日志表
DEFINE TABLE review_history SCHEMAFULL;

DEFINE FIELD task_id ON review_history TYPE record<review_tasks>;
DEFINE FIELD action ON review_history TYPE string
    ASSERT $value IN ['created', 'submitted', 'in_review', 'approved', 'rejected', 'cancelled', 'updated'];
DEFINE FIELD user_id ON review_history TYPE record<users>;
DEFINE FIELD user_name ON review_history TYPE string;
DEFINE FIELD comment ON review_history TYPE option<string>;
DEFINE FIELD changes ON review_history TYPE option<object>;
DEFINE FIELD timestamp ON review_history TYPE datetime DEFAULT time::now();

-- 索引
DEFINE INDEX review_history_task_idx ON review_history FIELDS task_id;
DEFINE INDEX review_history_timestamp_idx ON review_history FIELDS timestamp;
```

### 5. review_comments 表 - 批注评论

```surql
-- 批注评论表（多角色意见）
DEFINE TABLE review_comments SCHEMAFULL;

DEFINE FIELD annotation_id ON review_comments TYPE string ASSERT $value != NONE;
DEFINE FIELD annotation_type ON review_comments TYPE string
    ASSERT $value IN ['text', 'cloud', 'rect', 'obb'];
DEFINE FIELD author_id ON review_comments TYPE record<users>;
DEFINE FIELD author_name ON review_comments TYPE string;
DEFINE FIELD author_role ON review_comments TYPE string;
DEFINE FIELD content ON review_comments TYPE string ASSERT $value != NONE;
DEFINE FIELD reply_to_id ON review_comments TYPE option<record<review_comments>>;
DEFINE FIELD created_at ON review_comments TYPE datetime DEFAULT time::now();
DEFINE FIELD updated_at ON review_comments TYPE option<datetime>;

-- 索引
DEFINE INDEX review_comments_annotation_idx ON review_comments FIELDS annotation_id, annotation_type;
DEFINE INDEX review_comments_author_idx ON review_comments FIELDS author_id;
```

### 6. review_attachments 表 - 附件

```surql
-- 附件表
DEFINE TABLE review_attachments SCHEMAFULL;

DEFINE FIELD task_id ON review_attachments TYPE record<review_tasks>;
DEFINE FIELD name ON review_attachments TYPE string ASSERT $value != NONE;
DEFINE FIELD url ON review_attachments TYPE string ASSERT $value != NONE;
DEFINE FIELD size ON review_attachments TYPE option<int>;
DEFINE FIELD mime_type ON review_attachments TYPE option<string>;
DEFINE FIELD uploaded_by ON review_attachments TYPE record<users>;
DEFINE FIELD uploaded_at ON review_attachments TYPE datetime DEFAULT time::now();

-- 索引
DEFINE INDEX review_attachments_task_idx ON review_attachments FIELDS task_id;
```

## API 端点设计

### 编校审单 API

| 方法 | 端点 | 功能 |
|------|------|------|
| POST | `/api/review/tasks` | 创建编校审单 |
| GET | `/api/review/tasks` | 获取任务列表 |
| GET | `/api/review/tasks/:id` | 获取任务详情 |
| PATCH | `/api/review/tasks/:id` | 更新任务 |
| DELETE | `/api/review/tasks/:id` | 删除任务 |
| POST | `/api/review/tasks/:id/start-review` | 开始审核 |
| POST | `/api/review/tasks/:id/approve` | 通过审核 |
| POST | `/api/review/tasks/:id/reject` | 驳回审核 |
| POST | `/api/review/tasks/:id/cancel` | 取消任务 |
| GET | `/api/review/tasks/:id/history` | 获取历史 |

### 确认记录 API

| 方法 | 端点 | 功能 |
|------|------|------|
| POST | `/api/review/records` | 保存确认记录 |
| GET | `/api/review/records/:taskId` | 获取任务记录 |
| DELETE | `/api/review/records/:recordId` | 删除记录 |
| DELETE | `/api/review/records/task/:taskId` | 清空任务记录 |

### 评论 API

| 方法 | 端点 | 功能 |
|------|------|------|
| POST | `/api/review/comments` | 添加评论 |
| GET | `/api/review/comments/:annotationId` | 获取评论 |
| DELETE | `/api/review/comments/:commentId` | 删除评论 |

### 用户 API

| 方法 | 端点 | 功能 |
|------|------|------|
| GET | `/api/users` | 获取用户列表 |
| GET | `/api/users/me` | 获取当前用户 |
| GET | `/api/users/reviewers` | 获取审核人员 |

### WebSocket 端点

| 端点 | 功能 |
|------|------|
| `/ws/review` | 全局审核通知 |
| `/ws/review/user/:userId` | 用户专属通知 |

## WebSocket 消息类型

```typescript
type ReviewWebSocketMessageType =
  | 'task_created'      // 新任务创建
  | 'task_updated'      // 任务更新
  | 'task_submitted'    // 任务提交审核
  | 'task_approved'     // 任务通过
  | 'task_rejected'     // 任务驳回
  | 'task_cancelled'    // 任务取消
  | 'comment_added'     // 新评论
  | 'record_saved';     // 确认记录保存

type ReviewWebSocketMessage = {
  type: ReviewWebSocketMessageType;
  data: {
    taskId: string;
    task?: ReviewTask;
    userId?: string;
    userName?: string;
    comment?: string;
  };
  timestamp: string;
};
```
